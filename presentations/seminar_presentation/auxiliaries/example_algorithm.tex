                        [$c \aeq c$], $\emptyset$)} 
                        [.{unify($\emptyset$, $\{X \rightarrow c \}$, 
                            [], $\emptyset$)} 
                            [.{$(\emptyset, \{X \rightarrow c \}, \emptyset)$} ]
                        ]
                    ]
            ] 
       ]
}
